```kotlin
import java.lang.StringBuilder

val random = Random()

val hiragana = arrayOf(
        "あ", "い", "う", "え", "お",
        "か", "き", "く", "け", "こ",
        "さ", "し", "す", "せ", "そ",
        "た", "ち", "つ", "て", "と",
        "な", "に", "ぬ", "ね", "の",
        "は", "ひ", "ふ", "へ", "ほ",
        "ま", "み", "む", "め", "も",
        "や", "ゆ", "よ",
        "ら", "り", "る", "れ", "ろ",
        "わ", "を", "ん"
    )

 fun randomHiragana(length: Int): String {
        return StringBuilder().apply {

            for (i in 0 until length) {
                append(hiragana[random.nextInt(hiragana.size)])
            }

        }.toString()
    }
```
